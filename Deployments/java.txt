## STEPS TO DEPLOY A JAVA/SPRINGBOOT REST API ON A LINUX SERVER:

# GET UPDATES
sudo apt update

# INSTALL RELEVANT OPENJDK
sudo apt install openjdk-17-jdk

# INSTALL MAVEN 
wget https://mirrors.estointernet.in/apache/maven/maven-3/3.6.3/binaries/apache-maven-3.6.3-bin.tar.gz
tar -xvf apache-maven-3.6.3-bin.tar.gz
mv apache-maven-3.6.3 /opt/
M2_HOME='/opt/apache-maven-3.6.3'
PATH="$M2_HOME/bin:$PATH"
export PATH

# CLONE THE CODE FROM GITHUB
git clone https://github.com/GerromeSieger/RecipeApp-Java.git

# CHANGE DIRECTORY INTO THE CLONED REPO AND INSTALL DEPENDENCIES WITH MAVEN
cd RecipeApp-Java
mvn clean install

# CONFIRM DATABASE CONNECTION 

# RUN APPLICATION
mvn spring-boot:run
